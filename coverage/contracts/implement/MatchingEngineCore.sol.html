                        liquidityBitmap.toggleSingleBit(step.pipNext, false);

                        state.pip = step.pipNext;
                        state.isFullBuy = 0;
                        tickPosition[step.pipNext].fullFillLiquidity();

                singleSlot.isFullBuy = state.isFullBuy;
